    v_cost  := nullif(v_item->>'cost_cents','')::int;

    set
      price_cents = coalesce(v_price, price_cents),
      cost_cents  = coalesce(v_cost,  cost_cents)
    where id = v_product_id;

    if found then
      v_count := v_count + 1;
    end if;
  end loop;

  return v_count;
end;

declare
  r record;
begin
  if not public.is_admin() then
    raise exception 'Not authorized';
  end if;

  -- check stock
  for r in
    select product_id, qty
    from order_items
    where order_id = p_order_id
  loop
    if (select stock_qty from products where id = r.product_id) < r.qty then
      raise exception 'Insufficient stock for product %', r.product_id;
    end if;
  end loop;

  -- deduct stock
  update products p
  set stock_qty = stock_qty - oi.qty
  from order_items oi
  where oi.order_id = p_order_id
    and oi.product_id = p.id;

  -- confirm order
  update orders
  set status = 'confirmed'
  where id = p_order_id;
end;

| public      | apply_payment_status_rules     |                                                                                                                                                                                                                                                                                                                                          | trigger                                                                                             | CREATE OR REPLACE FUNCTION public.apply_payment_status_rules()
 RETURNS trigger
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
declare
  ref_clean text;
begin
  -- Normalize reference
  ref_clean := nullif(btrim(coalesce(new.reference_number, '')), '');
  -- allow gcash_ref too if you prefer:
  -- ref_clean := coalesce(ref_clean, nullif(btrim(coalesce(new.gcash_ref, '')), ''));

  -- COD: treat as paid right away (your chosen business rule)
  if new.method::text = 'cod' then
    new.status = 'paid'::payment_status_type;
    new.paid_at = coalesce(new.paid_at, now());
    return new;
  end if;

  -- GCash: pending unless VERIFIED by admin
  if new.method::text = 'gcash' then
    -- No ref or TO-FOLLOW => pending
    if ref_clean is null
       or upper(ref_clean) in ('TO-FOLLOW', 'TO FOLLOW', 'TOFOLLOW') then
      new.status = 'pending'::payment_status_type;
      new.paid_at = null;
      return new;
    end if;

    -- Has a reference number:
    -- If someone tries to mark paid, require verified_by (admin action)
    if new.status::text = 'paid' then
      if new.verified_by is null then
        new.status = 'pending'::payment_status_type;
        new.paid_at = null;
      else
        new.paid_at = coalesce(new.paid_at, now());
      end if;
    else
      -- default for gcash with ref is pending until verified
      new.status = 'pending'::payment_status_type;
      new.paid_at = null;
    end if;

    return new;
  end if;

  -- Other methods: leave as-is (or decide your own rule)
  return new;
end;

| public      | consume_inventory_fifo         | p_order_id uuid, p_product_id uuid, p_qty integer                                                                                                                                                                                                                                                                                        | integer                                                                                             | CREATE OR REPLACE FUNCTION public.consume_inventory_fifo(p_order_id uuid, p_product_id uuid, p_qty integer)
 RETURNS integer
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
declare
  v_needed int := p_qty;
  v_taken int;
  v_cost_sum bigint := 0;
  v_qty_sum bigint := 0;
  r record;
  v_unit_cost_fallback int;
begin
  if p_qty <= 0 then
    return 0;
  end if;

  -- fallback to product cost if no lots exist
  select cost_cents into v_unit_cost_fallback
  from public.products
  where id = p_product_id;

  for r in
    select id, qty_remaining, unit_cost_cents
    from public.inventory_lots
    where product_id = p_product_id
      and qty_remaining > 0
    order by created_at asc
    for update
  loop
    exit when v_needed <= 0;

    v_taken := least(v_needed, r.qty_remaining);

    update public.inventory_lots
    set qty_remaining = qty_remaining - v_taken
    where id = r.id;

    insert into public.order_item_lot_allocations
      (order_id, product_id, lot_id, qty, unit_cost_cents)
    values
      (p_order_id, p_product_id, r.id, v_taken, r.unit_cost_cents);

    v_cost_sum := v_cost_sum + (r.unit_cost_cents::bigint * v_taken::bigint);
    v_qty_sum := v_qty_sum + v_taken::bigint;
    v_needed := v_needed - v_taken;
  end loop;

  -- If we didn’t have enough lots, cost fallback for remainder (still track as "no lot")
  if v_needed > 0 then
    v_cost_sum := v_cost_sum + (coalesce(v_unit_cost_fallback, 0)::bigint * v_needed::bigint);
    v_qty_sum := v_qty_sum + v_needed::bigint;
    -- no allocation rows for fallback remainder; optional: you can insert a special lot row later
  end if;

  if v_qty_sum = 0 then
    return coalesce(v_unit_cost_fallback, 0);
  end if;

  return (v_cost_sum / v_qty_sum)::int;
end;

declare
  v_order_id uuid;
  v_order_code text;
  v_subtotal int := 0;
  v_delivery_fee int := 0;
  v_total int := 0;
  r record;
begin
  -- delivery fee (flat, configurable later)
  if p_fulfillment = 'delivery' then
    v_delivery_fee := 1000;
  end if;

  -- calculate subtotal from DB prices
  for r in
    select
      p.id,
      p.name,
      p.category,
      p.price_cents,
      p.cost_cents,
      (item->>'qty')::int as qty
    from jsonb_array_elements(p_items) item
    join products p on p.id = (item->>'product_id')::uuid
    where p.is_active = true
  loop
    v_subtotal := v_subtotal + (r.price_cents * r.qty);
  end loop;

  v_total := v_subtotal + v_delivery_fee;

  -- generate short order code
  v_order_code := 'ORD-' || upper(substr(gen_random_uuid()::text, 1, 6));

  -- create order
  insert into orders (
    order_code,
    customer_name,
    contact,
    notes,
    fulfillment,
    delivery_fee_cents,
    delivery_location,
    payment_method,
    subtotal_cents,
    total_cents
  )
  values (
    v_order_code,
    p_customer_name,
    p_contact,
    p_notes,
    p_fulfillment,
    v_delivery_fee,
    case when p_fulfillment = 'delivery' then p_delivery_location else null end,
    p_payment_method,
    v_subtotal,
    v_total
  )
  returning id into v_order_id;

  -- insert order items (snapshot)
  insert into order_items (
    order_id,
    product_id,
    name_snapshot,
    category_snapshot,
    unit_price_cents,
    unit_cost_cents,
    qty,
    line_total_cents
  )
  select
    v_order_id,
    p.id,
    p.name,
    p.category,
    p.price_cents,
    p.cost_cents,
    (item->>'qty')::int,
    p.price_cents * (item->>'qty')::int
  from jsonb_array_elements(p_items) item
  join products p on p.id = (item->>'product_id')::uuid;

  -- insert payment
  insert into payments (
    order_id,
    method,
    amount_cents,
    reference_number
  )
  values (
    v_order_id,
    p_payment_method,
    v_total,
    case when p_payment_method = 'gcash' then p_gcash_reference else null end
  );

  return query
  select v_order_id, v_order_code;
end;

    raise exception 'not authorized';

  into v_method, v_ref

  set status = 'paid',
      paid_at = coalesce(paid_at, now()),
      verified_by = coalesce(verified_by, auth.uid())
  where order_id = p_order_id
    and status <> 'paid';

  if not found then
    raise exception 'already paid';
  end if;

declare
  v_item record;
  v_total int := 0;
  v_subtotal int := 0;
  v_updated_products jsonb := '[]'::jsonb;
begin
  if p_order_id is null then
    raise exception 'order_id required';
  end if;

  if coalesce(trim(p_customer_name), '') = '' then
    raise exception 'customer_name required';
  end if;

  if p_items is null or jsonb_typeof(p_items) <> 'array' or jsonb_array_length(p_items) = 0 then
    raise exception 'items required';
  end if;

  -- 1) decrement stock
  for v_item in
    select
      (x->>'product_id')::uuid as product_id,
      coalesce((x->>'qty')::int, 0) as qty
    from jsonb_array_elements(p_items) as x
  loop
    if v_item.qty <= 0 then
      raise exception 'invalid qty for product %', v_item.product_id;
    end if;

    update public.products p
    set stock_qty = p.stock_qty - v_item.qty,
        updated_at = now()
    where p.id = v_item.product_id
      and p.is_active = true
      and p.stock_qty >= v_item.qty;

    if not found then
      raise exception 'out of stock (or inactive) for product %', v_item.product_id;
    end if;
  end loop;

  -- 2) insert order
  insert into public.orders (
    id, order_code, customer_name, contact, notes,
    fulfillment, pickup_location, delivery_fee_cents, delivery_location,
    payment_method, subtotal_cents, total_cents, status
  ) values (
    p_order_id,
    p_order_code,
    trim(p_customer_name),
    p_contact,
    nullif(trim(p_notes), ''),
    p_fulfillment,
    nullif(trim(p_pickup_location), ''),
    coalesce(p_delivery_fee_cents, 0),
    nullif(trim(p_delivery_location), ''),
    p_payment_method,
    0,
    0,
    'pending'::order_status_type
  );

  insert into public.order_items (
    order_id, product_id, name_snapshot, category_snapshot,
    unit_price_cents, unit_cost_cents, qty, line_total_cents
  )
  select
    p_order_id,
    p.id,
    p.name,
    p.category,
    p.price_cents,
    p.cost_cents,
    (x->>'qty')::int,
    ((x->>'qty')::int * p.price_cents)
  from jsonb_array_elements(p_items) x
  join public.products p on p.id = (x->>'product_id')::uuid;

  select coalesce(sum(line_total_cents), 0)::int
    into v_subtotal
  from public.order_items
  where order_id = p_order_id;

  v_total := v_subtotal + coalesce(p_delivery_fee_cents, 0);

      total_cents = v_total,
      updated_at = now()

    'total_cents', v_total,
    'updated_products', coalesce(v_updated_products, '[]'::jsonb)

 RETURNS jsonb
 LANGUAGE plpgsql
 SECURITY DEFINER
AS $function$
DECLARE
  v_batch_id uuid;
  v_batch_code text;
  v_item jsonb;
BEGIN
  -- 1. Insert the batch with structured data
  INSERT INTO public.inventory_batches (
    note, 
    supplier_name, 
    receipt_reference, 
    expiry_date, 
    created_by
  )
  VALUES (
    p_note, 
    p_supplier, 
    p_receipt, 
    p_expiry, 
    auth.uid()
  )
  RETURNING id, batch_code INTO v_batch_id, v_batch_code;

  -- 2. Loop through items to update stock and create lots
  -- Note: We use 'LOOP' and 'END LOOP' (no underscore)
  FOR v_item IN SELECT * FROM jsonb_array_elements(p_items)
  LOOP
    -- Insert inventory lot (Links batch to products)
    INSERT INTO public.inventory_lots (
      batch_id, 
      product_id, 
      qty_received, 
      qty_remaining, 
      unit_cost_cents
    )
    VALUES (
      v_batch_id, 
      (v_item->>'product_id')::uuid, 
      (v_item->>'qty')::int, 
      (v_item->>'qty')::int, 
      (v_item->>'unit_cost_cents')::bigint
    );

    -- Update main product stock total
    UPDATE public.products 
    SET stock_qty = COALESCE(stock_qty, 0) + (v_item->>'qty')::int
    WHERE id = (v_item->>'product_id')::uuid;
  END LOOP;

  RETURN jsonb_build_object('batch_id', v_batch_id, 'batch_code', v_batch_code);
END;

declare
  v_clean text;
begin
  if p_order_id is null then
    raise exception 'order_id required';
  end if;

  v_clean := nullif(trim(coalesce(p_suggestion, '')), '');

  -- allow "no suggestion" silently
  if v_clean is null then
    return;
  end if;

  if length(v_clean) > 500 then
    raise exception 'suggestion too long (max 500 chars)';
  end if;

  -- ensure order exists
  if not exists (select 1 from public.orders o where o.id = p_order_id) then
    raise exception 'order not found';
  end if;

  insert into public.order_suggestions (order_id, suggestion)
  values (p_order_id, v_clean)
  on conflict (order_id)
  do update set
    suggestion = excluded.suggestion,
    updated_at = now();

  return;
end;

| public      | sync_payment_for_order         | p_order_id uuid                                                                                                                                                                                                                                                                                                                          | void                                                                                                | CREATE OR REPLACE FUNCTION public.sync_payment_for_order(p_order_id uuid)
 RETURNS void
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
declare
  v_method payment_method_type;
  v_amount int;
  v_status payment_status_type;
  v_paid_at timestamptz;
  v_ref text;
begin
  -- pull current order snapshot
  select o.payment_method, o.total_cents
    into v_method, v_amount
  from public.orders o
  where o.id = p_order_id;

  if not found then
    raise exception 'order not found: %', p_order_id;
  end if;

  -- enforce payment rules:
  -- COD => paid
  -- GCash/others => pending (until verified)
  v_status :=
    case
      when v_method = 'cod'::payment_method_type then 'paid'::payment_status_type
      else 'pending'::payment_status_type
    end;

  v_paid_at :=
    case
      when v_status = 'paid'::payment_status_type then now()
      else null
    end;

  -- If you store GCash ref in payments.gcash_ref, keep whatever user submitted already.
  -- We do NOT overwrite it here unless you want to.
  insert into public.payments (
    order_id,
    method,
    amount_cents,
    status,
    paid_at
  )
  values (
    p_order_id,
    v_method,
    v_amount,
    v_status,
    v_paid_at
  )
  on conflict (order_id)
  do update set
    method = excluded.method,
    amount_cents = excluded.amount_cents,
    status = excluded.status,
    paid_at = case
      when excluded.status = 'paid'::payment_status_type then coalesce(public.payments.paid_at, excluded.paid_at)
      else public.payments.paid_at
    end;

end;

 RETURNS void
 LANGUAGE plpgsql
 SECURITY DEFINER
 SET search_path TO 'public'
AS $function$
declare
  v_method payment_method_type;
  v_total int;
  v_ref text;
  v_exists boolean;
begin
  select o.payment_method, o.total_cents
    into v_method, v_total
  from public.orders o
  where o.id = p_order_id;

  if v_method is null then
    raise exception 'order not found for %', p_order_id;
  end if;

  v_ref := nullif(trim(coalesce(p_payment_ref, '')), '');

  select exists(
    select 1 from public.payments p where p.order_id = p_order_id
  ) into v_exists;

  -- If we are ABOUT TO INSERT a NEW payment for gcash, ref must exist
  if (not v_exists) and v_method = 'gcash'::payment_method_type and v_ref is null then
    raise exception 'GCash reference number is required'
      using errcode = '23514';
  end if;

  -- Update existing payment row(s) if present
  update public.payments p
  set
    method = v_method,
    amount_cents = coalesce(v_total, 0),
    reference_number = case
      when v_method = 'gcash'::payment_method_type
        then coalesce(p.reference_number, v_ref)  -- keep existing, fill if missing
      else null
    end
  where p.order_id = p_order_id;

  if found then
    return;
  end if;

  -- Insert when none exists
  insert into public.payments (order_id, method, amount_cents, reference_number, status)
  values (
    p_order_id,
    v_method,
    coalesce(v_total, 0),
    case when v_method = 'gcash'::payment_method_type then v_ref else null end,
    'pending'::payment_status_type
  );
end;

declare
  v_has_payment boolean;
begin
  -- If the order is GCash:
  -- - If a payment row doesn't exist yet, DO NOTHING (avoid constraint violation).
  -- - If it already exists, sync totals/method without wiping reference_number.
  if new.payment_method = 'gcash'::payment_method_type then
    select exists(
      select 1 from public.payments p where p.order_id = new.id
    ) into v_has_payment;

    if not v_has_payment then
      return new;
    end if;

    perform public.sync_payment_for_order(new.id, null);
    return new;
  end if;

  -- For COD (and any non-gcash methods), normal behavior is fine.
  perform public.sync_payment_for_order(new.id, null);
  return new;
end;
